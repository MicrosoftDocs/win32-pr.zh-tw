```
propertyDescription
   name = System.Contact.OfficeLocation
   shellPKey = PKEY_Contact_OfficeLocation
   formatID = 176DC63C-2688-4E89-8143-A347800F25E9
   propID = 7
   SearchInfo
      InInvertedIndex = true
      IsColumn = true
   typeInfo
      type = String
```
