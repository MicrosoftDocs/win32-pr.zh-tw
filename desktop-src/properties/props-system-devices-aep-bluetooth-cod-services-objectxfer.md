## <a name="windows-10-version-1703-windows-10-version-1607-windows-10-version-1511"></a>Windows 10，1703版、Windows 10、1607版、Windows 10、1511版

```
propertyDescription
   name = System.Devices.Aep.Bluetooth.Cod.Services.ObjectXfer
   shellPKey = PKEY_Devices_Aep_Bluetooth_Cod_Services_ObjectXfer
   formatID = 5FBD34CD-561A-412E-BA98-478A6B0FEF1D
   propID = 9
   SearchInfo
      InInvertedIndex = false
      IsColumn = false

      type = Boolean
```
