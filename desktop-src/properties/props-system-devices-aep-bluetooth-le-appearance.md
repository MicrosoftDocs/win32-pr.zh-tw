```
propertyDescription
   name = System.Devices.Aep.Bluetooth.Le.Appearance
   shellPKey = PKEY_Devices_Aep_Bluetooth_Le_Appearance
   formatID = 995EF0B0-7EB3-4A8B-B9CE-068BB3F4AF69
```
