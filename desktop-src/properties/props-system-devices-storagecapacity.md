```
propertyDescription
   name = System.Devices.StorageCapacity
   shellPKey = PKEY_Devices_StorageCapacity
   formatID = 49CD1F76-5626-4B17-A4E8-18B4AA1A2213
   propID = 12
   SearchInfo
      InInvertedIndex = false
      IsColumn = false
   typeInfo
      type = UInt64
      IsInnate = true
      EnumeratedList
         UseValueForDefault = True
         enumRange
            name = StorageCapacity
```
