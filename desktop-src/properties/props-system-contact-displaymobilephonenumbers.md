   name = System.Contact.DisplayMobilePhoneNumbers
   shellPKey = PKEY_Contact_DisplayMobilePhoneNumbers
   formatID = 9CB0C358-9D7A-46B1-B466-DCC6F1A3D93D

   SearchInfo
