```
propertyDescription
   name = System.Calendar.RequiredAttendeeAddresses
   shellPKey = PKEY_Calendar_RequiredAttendeeAddresses
   formatID = 0BA7D6C3-568D-4159-AB91-781A91FB71E5
   propID = 100
```
