```cpp
  case CF_BITMAP:
        hdcMem = CreateCompatibleDC(hdc);
        if (hdcMem != NULL)
        {
            if (OpenClipboard(hwnd))
            {
                hbm = (HBITMAP) 
                     GetClipboardData(uFormat);
                SelectObject(hdcMem, hbm);
                GetClientRect(hwnd, &rc);
 
                BitBlt(hdc, 0, 0, rc.right, rc.bottom,
                    hdcMem, 0, 0, SRCCOPY);
                CloseClipboard();
             }
             DeleteDC(hdcMem);
        }
    break;
```
